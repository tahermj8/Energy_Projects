In [6]:
def vestas (wind):
        if wind < 4:
            return 0
        elif wind < 6:
            return 0.2
        elif wind < 8:
            return 0.5
        elif wind < 10:
            return 1.4
        elif wind < 12:
            return 2.6
        elif wind < 25: #cutoff speed
            return 3
        else:
            return 0
In [7]:
def enercon (wind):
        if wind < 4:
            return 0
        elif wind < 6:
            return 0.1
        elif wind < 8:
            return 0.3
        elif wind < 10:
            return 0.8
        elif wind < 12:
            return 1.6
        elif wind < 14:
            return 2.2
        elif wind < 16:
            return 2.7
        elif wind < 28:  #cutoff speed
            return 3
        else:
            return 0
